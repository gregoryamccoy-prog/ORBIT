# Satellite Visualization System

## Developer Task Breakdown + Proposed File Structure + Starter Code

---

## 1. Recommended Stack

**Language:** TypeScript
**Framework:** React
**Build Tool:** Vite
**3D Globe Engine:** CesiumJS
**Orbit Math / Propagation:** satellite.js
**State Management:** Zustand
**HTTP Client:** native `fetch`
**Styling:** CSS modules or plain CSS

### Why this stack

This is the shortest path to a polished result. Cesium already solves the hard Earth-rendering problem. satellite.js handles TLE propagation. React handles the UI cleanly. TypeScript keeps the codebase from turning into garbage after the first few feature additions.

---

## 2. Suggested Delivery Approach

Build in this order:

1. **Static globe renders in Cesium**
2. **Load a small fixed list of satellites**
3. **Render satellites as glyphs**
4. **Add click selection**
5. **Show info panel**
6. **Draw a single selected orbit track**
7. **Add Earth-side observer mode**
8. **Add minute-based updates**
9. **Add textbox count limit**
10. **Add API-backed metadata**
11. **Optimize toward 1000 satellites**

Do not start with metadata APIs or fancy imagery tweaks. Get the globe and selectable dots working first.

## 8. Orbit Service

## `src/services/orbitService.ts`

```ts
import * as satellite from "satellite.js";
import type { TleRecord, SatellitePosition } from "../types/satellite";

function safeDegrees(rad: number): number {
  return (rad * 180) / Math.PI;
}

export function computeSatellitePosition(record: TleRecord, now: Date): SatellitePosition | null {
  try {
    const satrec = satellite.twoline2satrec(record.line1, record.line2);
    const pv = satellite.propagate(satrec, now);

    if (!pv.position || !pv.velocity) {
      return null;
    }

    const gmst = satellite.gstime(now);
    const geodetic = satellite.eciToGeodetic(pv.position, gmst);

    const latitudeDeg = safeDegrees(geodetic.latitude);
    const longitudeDeg = safeDegrees(geodetic.longitude);
    const altitudeKm = geodetic.height;

    const vx = pv.velocity.x;
    const vy = pv.velocity.y;
    const vz = pv.velocity.z;
    const speedKps = Math.sqrt(vx * vx + vy * vy + vz * vz);

    return {
      id: record.id,
      name: record.name,
      latitudeDeg,
      longitudeDeg,
      altitudeKm,
      speedKps,
    };
  } catch (error) {
    console.warn(`Failed to compute position for satellite ${record.name}`, error);
    return null;
  }
}

export function computeSatellitePositions(records: TleRecord[], now: Date): SatellitePosition[] {
  return records
    .map((record) => computeSatellitePosition(record, now))
    .filter((item): item is SatellitePosition => item !== null);
}
```

const metadataCache = new Map<string, SatelliteMetadata>();

export async function getSatelliteMetadata(position: SatellitePosition): Promise<SatelliteMetadata> {
  const cached = metadataCache.get(position.id);

    return cached;

## 12. Update Hook

## `src/hooks/useSatelliteUpdater.ts`

```ts
import { useEffect } from "react";
import { POSITION_REFRESH_MS } from "../config/constants";
import { computeSatellitePositions } from "../services/orbitService";
import { useAppStore } from "../state/appStore";

export function useSatelliteUpdater(): void {
  const tleRecords = useAppStore((s) => s.tleRecords);
  const satelliteLimit = useAppStore((s) => s.satelliteLimit);
  const setPositions = useAppStore((s) => s.setPositions);

  useEffect(() => {
    function updateNow(): void {
      const activeRecords = tleRecords.slice(0, satelliteLimit);
      const positions = computeSatellitePositions(activeRecords, new Date());
      setPositions(positions);
    }

    updateNow();
    const timer = window.setInterval(updateNow, POSITION_REFRESH_MS);

    return () => {
      window.clearInterval(timer);
    };
  }, [tleRecords, satelliteLimit, setPositions]);
}
```

```tsx
import { ChangeEvent } from "react";
import { MAX_SATELLITE_LIMIT, MIN_SATELLITE_LIMIT } from "../config/constants";
import { useAppStore } from "../state/appStore";

export function ControlPanel(): JSX.Element {
  const satelliteLimit = useAppStore((s) => s.satelliteLimit);
  const setSatelliteLimit = useAppStore((s) => s.setSatelliteLimit);
  const viewMode = useAppStore((s) => s.viewMode);
  const setViewMode = useAppStore((s) => s.setViewMode);

  function handleLimitChange(event: ChangeEvent<HTMLInputElement>): void {
    const raw = Number(event.target.value);

    if (Number.isNaN(raw)) {
      return;
    }

    const clamped = Math.max(MIN_SATELLITE_LIMIT, Math.min(MAX_SATELLITE_LIMIT, raw));
    setSatelliteLimit(clamped);
  }

  return (
    <div className="control-panel">
      <label>
        Satellites:
        <input
          type="number"
          min={MIN_SATELLITE_LIMIT}
          max={MAX_SATELLITE_LIMIT}
          value={satelliteLimit}
          onChange={handleLimitChange}
        />
      </label>

      <button
        type="button"
        onClick={() => setViewMode("earthside")}
        disabled={viewMode === "earthside"}
      >
        Crestwood View
      </button>

      <button
        type="button"
        onClick={() => setViewMode("orbital")}
        disabled={viewMode === "orbital"}
      >
        Orbital View
      </button>
    </div>
  );
}
```

## 15. Cesium Globe Component

This is the core. Keep it clean.

## `src/components/CesiumGlobe.tsx`

```tsx
import { useEffect, useRef } from "react";
import * as Cesium from "cesium";
import { flyToEarthSideObserverView, flyToOrbitalView } from "../services/cameraService";
import { getSatelliteMetadata } from "../services/metadataService";
import { useAppStore } from "../state/appStore";

export function CesiumGlobe(): JSX.Element {
  const containerRef = useRef<HTMLDivElement | null>(null);
  const viewerRef = useRef<Cesium.Viewer | null>(null);
  const orbitEntityRef = useRef<Cesium.Entity | null>(null);

  const positions = useAppStore((s) => s.positions);
  const viewMode = useAppStore((s) => s.viewMode);
  const setSelectedSatellite = useAppStore((s) => s.setSelectedSatellite);
  const setSelectedMetadata = useAppStore((s) => s.setSelectedMetadata);
  const setIsMetadataLoading = useAppStore((s) => s.setIsMetadataLoading);

  useEffect(() => {
    if (!containerRef.current || viewerRef.current) {
      return;
    }

    const viewer = new Cesium.Viewer(containerRef.current, {
      timeline: false,
      animation: false,
      baseLayerPicker: false,
      geocoder: false,
      homeButton: false,
      sceneModePicker: false,
      navigationHelpButton: false,
      infoBox: false,
      selectionIndicator: false,
    });

    viewer.scene.globe.enableLighting = true;
    viewerRef.current = viewer;
    flyToOrbitalView(viewer);

    const handler = new Cesium.ScreenSpaceEventHandler(viewer.scene.canvas);
    handler.setInputAction(async (click: Cesium.ScreenSpaceEventHandler.PositionedEvent) => {
      const picked = viewer.scene.pick(click.position);

      if (!Cesium.defined(picked) || !(picked as any).id) {
        return;
      }

      const entity = (picked as any).id as Cesium.Entity;
      const satId = entity.properties?.getValue(Cesium.JulianDate.now())?.satelliteId;
      const satName = entity.properties?.getValue(Cesium.JulianDate.now())?.satelliteName;

      if (!satId || !satName) {
        return;
      }

      setSelectedSatellite({ id: satId, name: satName });

      const matched = positions.find((p) => p.id === satId);
      if (!matched) {
        return;
      }

      setIsMetadataLoading(true);
      const metadata = await getSatelliteMetadata(matched);
      setSelectedMetadata(metadata);
      setIsMetadataLoading(false);

      if (orbitEntityRef.current) {
        viewer.entities.remove(orbitEntityRef.current);
        orbitEntityRef.current = null;
      }

      orbitEntityRef.current = viewer.entities.add({
        polyline: {
          positions: buildSimpleOrbitPolyline(matched.longitudeDeg, matched.latitudeDeg, matched.altitudeKm),
          width: 2,
          material: Cesium.Color.YELLOW,
        },
      });
    }, Cesium.ScreenSpaceEventType.LEFT_CLICK);

    return () => {
      handler.destroy();
      viewer.destroy();
      viewerRef.current = null;
    };
  }, [positions, setSelectedMetadata, setSelectedSatellite, setIsMetadataLoading]);

  useEffect(() => {
    const viewer = viewerRef.current;
    if (!viewer) {
      return;
    }

    viewer.entities.removeAll();

    for (const sat of positions) {
      viewer.entities.add({
        position: Cesium.Cartesian3.fromDegrees(sat.longitudeDeg, sat.latitudeDeg, sat.altitudeKm * 1000),
        point: {
          pixelSize: 6,
          color: Cesium.Color.CYAN,
        },
        properties: {
          satelliteId: sat.id,
          satelliteName: sat.name,
        },
      });
    }
  }, [positions]);

  useEffect(() => {
    const viewer = viewerRef.current;
    if (!viewer) {
      return;
    }

    if (viewMode === "earthside") {
      flyToEarthSideObserverView(viewer);
    } else {
      flyToOrbitalView(viewer);
    }
  }, [viewMode]);

  return <div ref={containerRef} className="cesium-container" />;
}

function buildSimpleOrbitPolyline(longitudeDeg: number, latitudeDeg: number, altitudeKm: number): Cesium.Cartesian3[] {
  const points: Cesium.Cartesian3[] = [];

  for (let i = 0; i <= 180; i += 5) {
    const lon = ((longitudeDeg + i * 2) % 360) - 180;
    const lat = latitudeDeg;
    points.push(Cesium.Cartesian3.fromDegrees(lon, lat, altitudeKm * 1000));
  }

  return points;
}
```

### Important note

That orbit track function is a placeholder. It is **not** a true propagated orbit path. Good enough for scaffolding, not for final. Replace it later with sampled propagation points over time.
